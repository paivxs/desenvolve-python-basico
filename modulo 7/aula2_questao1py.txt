# Atividade 1 - Conversão de data com mês por extenso

# lista com os nomes dos meses
meses = ["Janeiro", "Fevereiro", "Março", "Abril", "Maio", "Junho",
"Julho", "Agosto", "Setembro", "Outubro", "Novembro", "Dezembro"]

# Solicita a data
data = input("Digite uma data de nascimento: ")

# separa a data
dia, mes, ano = data.split("/")

# converte o numero do mês para o nome ( subtrai 1 pois listas começam em zero)
mes_extenso = meses[int(mes) - 1]

# imprime a data formatada
print(f"Você nasceu em {dia} de {mes_extenso} de {ano}.")
